concatena([], A, A).

concatena([H|T], B, [H|L]) :-
    concatena(T, B, L)

rivolta([], []).

rivolta([H|T], L) :-
    rivolta(T, L),
    concatena(T, H, L).


subtract(_, [], []).

subtract(X, [X|T], T).

subtract(X, [A|PT1], [A|PT2]) :-
    subtract(X, PT1, PT2).
    

permutazione([], []).

pertmutazione([H|T], B):-
    permutazione(T, PT1_2),
    appartiene(H, B),
    subtract(H, B, PT1_2).
